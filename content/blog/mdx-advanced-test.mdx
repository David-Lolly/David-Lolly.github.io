export const InteractiveCounter = () => {
  const [count, setCount] = useState(0)
  return (
    <div style={{
      padding: '20px',
      backgroundColor: 'rgb(var(--color-card))',
      borderRadius: '12px',
      border: '2px solid rgb(var(--color-primary))',
      margin: '20px 0',
      textAlign: 'center'
    }}>
      <h3 style={{ marginBottom: '15px', color: 'rgb(var(--color-primary))' }}>
        🎯 交互式计数器组件
      </h3>
      <div style={{ fontSize: '48px', fontWeight: 'bold', margin: '20px 0' }}>
        {count}
      </div>
      <div style={{ display: 'flex', gap: '10px', justifyContent: 'center' }}>
        <button
          onClick={() => setCount(count - 1)}
          style={{
            padding: '10px 20px',
            backgroundColor: 'rgb(var(--color-primary))',
            color: 'white',
            border: 'none',
            borderRadius: '6px',
            cursor: 'pointer',
            fontSize: '16px',
            fontWeight: '600'
          }}
        >
          - 减少
        </button>
        <button
          onClick={() => setCount(0)}
          style={{
            padding: '10px 20px',
            backgroundColor: 'rgb(var(--color-muted))',
            color: 'rgb(var(--color-foreground))',
            border: '1px solid rgb(var(--color-border))',
            borderRadius: '6px',
            cursor: 'pointer',
            fontSize: '16px',
            fontWeight: '600'
          }}
        >
          重置
        </button>
        <button
          onClick={() => setCount(count + 1)}
          style={{
            padding: '10px 20px',
            backgroundColor: 'rgb(var(--color-primary))',
            color: 'white',
            border: 'none',
            borderRadius: '6px',
            cursor: 'pointer',
            fontSize: '16px',
            fontWeight: '600'
          }}
        >
          + 增加
        </button>
      </div>
    </div>
  )
}

# MDX 是什么？

MDX 允许你在 Markdown 文档中使用 <Highlight color="#a7f3d0">JSX 和 React 组件</Highlight>，让你的文档更加强大和交互式！

## 1️⃣ 内联 React 组件

你可以像这样使用自定义高亮组件：

- 这是<Highlight>默认黄色高亮</Highlight>文本
- 这是<Highlight color="#fbbf24">橙色高亮</Highlight>文本
- 这是<Highlight color="#86efac">绿色高亮</Highlight>文本
- 这是<Highlight color="#93c5fd">蓝色高亮</Highlight>文本

## 2️⃣ 交互式组件

下面是一个完全可交互的 React 组件，尝试点击按钮：

<InteractiveCounter />

> **💡 提示：** 这个计数器完全运行在客户端，拥有完整的 React 状态管理！

    return fibonacci(n-1) + fibonacci(n-2)

## 6️⃣ 表格与列表

| 功能 | Markdown | MDX |
| :--- | :---: | :---: |
| 基础语法 | ✅ | ✅ |

使用列表展示 MDX 的优势：

- ✨ **组件化**：可以复用 React 组件
